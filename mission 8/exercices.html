8. Compter les syllabes 
        function compterSyllabes() {
            const texte = prompt("Entrez un texte :").toLowerCase();
            const syllabes = texte.match(/[aeiouy]+/g);  // On compte chaque groupe de voyelles comme une syllabe
            alert(`Nombre de syllabes : ${syllabes ? syllabes.length : 0}`);
        }
9. Calculer l'IMC 
        function calculerIMC() {
            const poids = parseFloat(prompt("Entrez votre poids en kg :"));
            const taille = parseFloat(prompt("Entrez votre taille en mètres :"));
            const imc = (poids / (taille * taille)).toFixed(2);
            alert(`Votre IMC est : ${imc}`);
        }
9. Générer un mot de passe aléatoire 
        function genererMotDePasse() {
            const longueur = parseInt(prompt("Entrez la longueur du mot de passe :"));
            if (isNaN(longueur) || longueur <= 0) {
                alert("Veuillez entrer un nombre valide pour la longueur.");
                return;
            }
        
            const caracteres = 'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789!@#$%^&*()';
            let motDePasse = '';
            for (let i = 0; i < longueur; i++) {
                const index = Math.floor(Math.random() * caracteres.length);
                motDePasse += caracteres.charAt(index);
            }
            alert(`Votre mot de passe généré est : ${motDePasse}`);
        }
10. Générer une séquence de Fibonacci 
        function genererFibonacci() {
            const n = parseInt(prompt("Entrez le nombre de termes de la séquence de Fibonacci :"));
            if (isNaN(n) || n <= 0) {
                alert("Veuillez entrer un nombre valide pour les termes.");
                return;
            }
            
            let fibonacci = [0, 1];
            for (let i = 2; i < n; i++) {
                fibonacci.push(fibonacci[i - 1] + fibonacci[i - 2]);
            }
            
            alert(`La séquence de Fibonacci avec ${n} termes est : ${fibonacci.join(", ")}`);
        }
